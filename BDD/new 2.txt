CREATE TABLE vendedor(id_vend VARCHAR(7) NOT NULL UNIQUE,
nombre_vend VARCHAR(15),
apepat_vend VARCHAR(15),
apemat_vend VARCHAR(15),
tel_vend VARCHAR(10),
mail_vend VARCHAR(20),
contra_vend VARCHAR(20),                      
INDEX(mail_vend),
PRIMARY KEY(id_vend));
